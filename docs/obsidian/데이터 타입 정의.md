# 데이터 타입 정의

> 경로: `src/types/`

## 사용자/인증

```typescript
type UserRole = 'admin' | 'hr_manager' | 'dept_manager' | 'employee'

interface Session {
  user_id: string
  user_name: string
  role: UserRole
  session_id: string
}
```

## 인사 기본

```typescript
interface Employee {
  id: string
  employee_number: string
  name: string
  name_en?: string
  email: string
  phone: string
  birth_date: string
  gender: 'male' | 'female'
  address: string
  address_detail?: string
  zip_code?: string
  department_id: string
  position_rank_id: string
  position_title_id: string
  employment_type: 'regular' | 'contract' | 'part_time' | 'intern'
  hire_date: string
  resignation_date?: string
  status: 'active' | 'on_leave' | 'resigned'
  base_salary: number
  bank_name?: string
  bank_account?: string
  profile_image_url?: string
  emergency_contact_name?: string
  emergency_contact_phone?: string
  emergency_contact_relation?: string
}

interface Department {
  id: string
  name: string
  code: string
  parent_id: string | null
  level: number
  sort_order: number
  is_active: boolean
  effective_from?: string
  effective_to?: string
}

interface PositionRank {
  id: string
  name: string
  level: number
  is_active: boolean
  effective_from?: string
  effective_to?: string
}

interface PositionTitle {
  id: string
  name: string
  level: number
  is_active: boolean
  effective_from?: string
  effective_to?: string
}
```

## 인사 부가정보

```typescript
interface CareerHistory {
  id: string
  employee_id: string
  company_name: string
  department: string
  position: string
  start_date: string
  end_date: string
  description?: string
}

interface EducationHistory {
  id: string
  employee_id: string
  school_name: string
  major: string
  degree: string
  start_date: string
  end_date: string
  is_graduated: boolean
}

interface Certification {
  id: string
  employee_id: string
  name: string
  issuer: string
  issue_date: string
  expiry_date?: string
  certificate_number?: string
}

interface FamilyMember {
  id: string
  employee_id: string
  name: string
  relation: string
  birth_date: string
  phone?: string
  is_dependent: boolean
}
```
